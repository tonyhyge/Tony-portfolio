    ```typescript
    "use client";

    import { Moon, Sun } from "lucide-react";
    import { Button } from "@/components/ui/button";
    import { useEffect, useState } from "react";

    export function ThemeToggle() {
      const [mounted, setMounted] = useState(false);

      useEffect(() => {
        setMounted(true);
      }, []);

      const toggle = () => {
        const isDark = document.documentElement.classList.toggle("dark");
        localStorage.setItem("theme", isDark ? "dark" : "light");
      };

      if (!mounted) {
        return <div className="size-8" aria-hidden="true" />;
      }

      return (
        <Button variant="ghost" size="icon" onClick={toggle} aria-label="Toggle theme">
          {document.documentElement.classList.contains("dark")
            ? <Sun className="size-4" />
            : <Moon className="size-4" />
          }
        </Button>
      );
    }
    ```

    Key design decisions (per CONTEXT.md D-03, D-11, and RESEARCH.md):
    - "use client" directive required for browser APIs (localStorage, classList)
    - `useState(false)` + `useEffect(() => setMounted(true), [])` prevents hydration mismatch (renders placeholder `<div>` same size as icon button before mount)
    - Direct DOM manipulation (`classList.toggle`) — no React context provider wrapper (avoids forcing entire tree client-side, per RESEARCH.md anti-pattern #1)
    - `localStorage key: "theme"` — matches the flash prevention script in layout.tsx (per CONTEXT.md)
    - Sun icon shown in dark mode (implies "switch to light"), Moon icon shown in light mode (implies "switch to dark") — standard UX pattern
    - Uses shadcn Button variant="ghost" size="icon" — matches existing component inventory (per CONTEXT.md)
  </action>

    export function NavLinks({ links, className }: NavLinksProps) {
      return (
        <nav className={cn("flex items-center gap-1", className)} aria-label="Main navigation">

              </a>
            ) : (
              <Link
                key={link.href}
                href={link.href}
                className="inline-flex h-9 w-max items-center justify-center rounded-lg px-2.5 py-1.5 text-sm font-medium transition-all hover:bg-muted focus-visible:ring-3 focus-visible:ring-ring/50"
