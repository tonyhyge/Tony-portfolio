    **File 1:** `src/hooks/useScrollDirection.ts`

    A custom hook that detects scroll direction for hide-on-scroll-down behavior. Must be "use client". Follows the hydration-safe pattern from ThemeToggle:

    ```typescript
    "use client";

    import { useState, useEffect, useCallback } from "react";

    export interface ScrollDirection {
      direction: "up" | "down";
      scrolled: boolean;
    }

    export function useScrollDirection(threshold = 10): ScrollDirection {
      const [scrollY, setScrollY] = useState(0);
      const [direction, setDirection] = useState<"up" | "down">("up");
      const [scrolled, setScrolled] = useState(false);

      const handleScroll = useCallback(() => {
        const current = window.scrollY;
        setScrollY(current);
        setScrolled(current > 50);

        const delta = current - scrollY;
        if (Math.abs(delta) > threshold) {
          setDirection(delta > 0 ? "down" : "up");
        }
      }, [scrollY, threshold]);

      useEffect(() => {
        window.addEventListener("scroll", handleScroll, { passive: true });
        return () => window.removeEventListener("scroll", handleScroll);
      }, [handleScroll]);

      return { direction, scrolled };
    }
    ```

    Per D-01 (hide-on-scroll-down), Claude's discretion (custom hook preferred for reuse), RESEARCH.md Pitfall 4 (10px threshold prevents jank). `handleScroll` uses `useCallback` with `passive: true` for performance (per RESEARCH.md anti-patterns). `scrolled` threshold at 50px per D-02 (backdrop fade at ~50px).

    **File 2:** `src/hooks/useActiveSection.ts`

    A custom hook using IntersectionObserver to track which section is currently in view. Observes all section elements by their IDs, returns the ID of the most recently intersected section.

    ```typescript
    "use client";

    import { useState, useEffect } from "react";

    export function useActiveSection(sectionIds: string[]): string {
      const [activeSection, setActiveSection] = useState(sectionIds[0] ?? "");

      useEffect(() => {
        if (sectionIds.length === 0) return;

        const observer = new IntersectionObserver(
          (entries) => {
            entries.forEach((entry) => {
              if (entry.isIntersecting) {
                setActiveSection(entry.target.id);
              }
            });
          },
          {
            rootMargin: "-20% 0px -60% 0px",
            threshold: [0, 0.25, 0.5],
          }
        );

        for (const id of sectionIds) {
          const el = document.getElementById(id);
          if (el) observer.observe(el);
        }

        return () => {
          observer.disconnect();
        };
      }, [sectionIds]);

      return activeSection;
    }
    ```

    Per D-05 (IntersectionObserver for active section), D-02 (dot indicator beside active link), Claude's discretion on rootMargin values. Default activeSection is `sectionIds[0]` per RESEARCH.md Pitfall 3 (ensures valid default on page load). Observer re-creates when `sectionIds` changes (e.g. if sections are added dynamically).
