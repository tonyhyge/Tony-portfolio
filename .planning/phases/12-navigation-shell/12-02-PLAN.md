    ```typescript
    "use client";

    import { ReactLenis } from "lenis/react";
    import "lenis/dist/lenis.css";

    interface LenisProviderProps {
      children: React.ReactNode;
    }

    export function LenisProvider({ children }: LenisProviderProps) {
      const prefersReducedMotion =
        typeof window !== "undefined"
          ? window.matchMedia("(prefers-reduced-motion: reduce)").matches
          : false;

      return (
        <ReactLenis
          root
          options={{
            duration: prefersReducedMotion ? 0 : 1.2,
            easing: (t: number) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
            smoothWheel: true,
            anchors: true,
            autoRaf: true,
          }}
        >
          {children}
        </ReactLenis>
      );
    }
    ```

    Per RESEARCH.md Pattern 1 (LenisProvider offical pattern). `root: true` attaches to `<html>` scroll container per Lenis React docs. `duration: 1.2` and easing function from RESEARCH.md standard settings. `prefersReducedMotion` detected via `window.matchMedia("(prefers-reduced-motion: reduce)")` — sets `duration: 0` to disable smooth scroll per RESEARCH.md Pitfall 6. `import "lenis/dist/lenis.css"` imports required Lenis CSS (sets `scroll-behavior: auto` natively, but globals.css already has this as backup).

    A Sheet-based mobile navigation. Per D-06 (slide-in from right using existing sheet.tsx), D-07 (auto-close on nav link tap). CRITICAL: Base UI Dialog.Close uses `render` prop (NOT `asChild`) per RESEARCH.md Pitfall 5.

    ```typescript
    "use client";

    import { useState } from "react";
    import Link from "next/link";
    import { Menu } from "lucide-react";
    import { Button } from "@/components/ui/button";
    import {
      Sheet,
      SheetTrigger,
      SheetContent,
      SheetClose,
    } from "@/components/ui/sheet";
    import { navLinks } from "@/app/data/navigation";

    export function MobileNav() {
      const [open, setOpen] = useState(false);

      return (
        <Sheet open={open} onOpenChange={setOpen}>
          <SheetTrigger>
            <Button variant="ghost" size="icon" aria-label="Toggle mobile menu" className="lg:hidden">
              <Menu className="size-5" />
            </Button>
          </SheetTrigger>
          <SheetContent side="right" showCloseButton={false}>
            <div className="flex flex-col gap-1 p-4 pt-12">
              <nav aria-label="Mobile navigation" className="flex flex-col gap-1">
                {navLinks.map((link) =>
                  link.isAnchor ? (
                    <SheetClose
                      key={link.href}
                      render={
                        <a
                          href={link.href}
                          className="flex min-h-[44px] items-center rounded-lg px-4 py-3 text-sm font-medium text-foreground transition-colors hover:bg-muted focus-visible:ring-3 focus-visible:ring-ring/50"
                        >
                          {link.label}
                        </a>
                      }
                    />
                  ) : (
                    <SheetClose
                      key={link.href}
                      render={
                        <Link
                          href={link.href}
                          className="flex min-h-[44px] items-center rounded-lg px-4 py-3 text-sm font-medium text-foreground transition-colors hover:bg-muted focus-visible:ring-3 focus-visible:ring-ring/50"
                          onClick={() => setOpen(false)}
                        >
                          {link.label}
                        </Link>
                      }
                    />
                  )
                )}
              </nav>
            </div>
          </SheetContent>
        </Sheet>
      );
    }
    ```

    Composes the existing StickyHeader shell with NavLinks, ThemeToggle, and MobileNav trigger. Per D-03 (uses existing sticky-header.tsx as base), D-01 (hide-on-scroll-down/show-on-scroll-up), D-02 (backdrop fade at ~50px).

    ```typescript
    "use client";

    import { cn } from "@/lib/utils";
    import { StickyHeader } from "./sticky-header";
    import { NavLinks } from "./nav-links";
    import { ThemeToggle } from "./theme-toggle";
    import { MobileNav } from "./mobile-nav";
    import { useScrollDirection } from "@/hooks/useScrollDirection";
    import { useActiveSection } from "@/hooks/useActiveSection";
    import { navLinks } from "@/app/data/navigation";

    export function SiteHeader() {
      const { direction, scrolled } = useScrollDirection();

      const sectionIds = navLinks.filter((l) => l.isAnchor).map((l) => l.href.replace("#", ""));
      const activeSection = useActiveSection(sectionIds);

      const isVisible = direction === "up" || !scrolled;

      return (
        <div
          className={cn(
            "fixed top-0 left-0 right-0 z-40 transition-transform duration-200 ease-in-out",
            isVisible ? "translate-y-0" : "-translate-y-full",
          )}
        >
          <StickyHeader
            className={cn(
              "transition-all duration-200",
              scrolled
                ? "bg-background/80 backdrop-blur-md border-b border-border/50"
                : "bg-transparent border-b border-transparent",
            )}
          >
            <div className="mx-auto flex h-14 max-w-screen-xl items-center justify-between px-4 sm:px-6 lg:px-8">
              <NavLinks
                links={navLinks}
                activeHref={activeSection ? `#${activeSection}` : undefined}
                className="hidden lg:flex"
              />
              <div className="flex items-center gap-2">
                <ThemeToggle />
                <MobileNav />
              </div>
            </div>
          </StickyHeader>
        </div>
      );
    }
    ```

    Key implementation details (cross-referencing UI-SPEC.md interaction states table):

    1. **Outer `<div>`** uses `fixed` positioning (not `sticky` like StickyHeader) because hide-on-scroll-down with `translateY` requires `fixed`. The StickyHeader's `sticky top-0` is overridden by the outer div's `fixed` positioning — this is correct: the outer div handles hide/show via `-translate-y-full` while StickyHeader provides the glassmorphism styling.

    2. **Hide/show state** per UI-SPEC.md: `isVisible` checks `direction === "up" || !scrolled` — header visible when scrolling up or at top of page, hidden when scrolling down past threshold. `transition-transform duration-200 ease-in-out` on outer div for smooth hide/show animation per UI-SPEC.md header hide/show state (uses `transform`, not `top`/`margin` per RESEARCH.md anti-pattern).

    3. **Backdrop state** per UI-SPEC.md: when `scrolled` is true (scrollY > 50), StickyHeader receives glassmorphism classes (`bg-background/80 backdrop-blur-md border-b border-border/50`). When at top, `bg-transparent border-b border-transparent`. `transition-all duration-200` for smooth backdrop fade.

    4. **NavLinks** receives `links={navLinks}` and is `hidden` at `lg:` breakpoint — desktop nav visible at >= 1024px per UI-SPEC.md responsive contract.

    5. **Right side** groups ThemeToggle and MobileNav trigger. ThemeToggle is hydration-safe (renders placeholder size-8 before mount). MobileNav trigger is hidden at `lg:` via its own `lg:hidden` class.

    6. **Inner container**: `mx-auto flex h-14 max-w-screen-xl items-center justify-between px-4 sm:px-6 lg:px-8` — matches UI-SPEC.md header spacing: `h-14`, `px-4 sm:px-6 lg:px-8`.
